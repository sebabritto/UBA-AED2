\section{Ejercicio 17: Arreglo de enteros no repetidos}
\textbf{Cumplen:}
\begin{lstlisting}
A = [1, 2, 3, 4, 5]
A = [2, 1, 3, 4, 5]
A = [2, 3, 1, 4, 5]
A = [2, 3, 4, 5, 1]
A = [2, 4, 3, 6, 5]
\end{lstlisting}

\begin{algorithm}[H]
\caption{
    \textbf{elementosMasChicosQuePos}(\textbf{in} r: recibe) $\to$ \textbf{out} d: devuelve
}
    \begin{algorithmic}[1]
        \State i $\gets$ tam(A)
        \While{i > 1}
            \If{A[i] < A[i-1]}
                \State Swap(A[i], A[i-1])
            \EndIf
            \State i $\gets$ i - 1
        \EndWhile
    \end{algorithmic}
    \Complexity{$O(n^2)$}
\end{algorithm}
